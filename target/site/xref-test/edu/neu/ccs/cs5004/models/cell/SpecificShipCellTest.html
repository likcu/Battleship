28    @Before
29    public void setUp() throws Exception {
30  
31      battleship = new Battleship(4, 4);
32      battleship2 = new Battleship(4, 0);
33      battleship3 = new Battleship(4, 1);
34      specificShipCell = new SpecificShipCell(true, true, battleship);
35      specificShipCell2 = new SpecificShipCell(true, true, battleship);
36      specificShipCell3 = new SpecificShipCell(true, false, battleship);
37      specificShipCell4 = new SpecificShipCell(false, false, battleship2);
38      specificShipCell5 = new SpecificShipCell(true, false, battleship3);
39      diffSpecificShipCell = new SpecificShipCell(true, false, battleship);
40      sameRefSpecificShipCell = specificShipCell;
41      sameStateAsSpecificShipCell = new SpecificShipCell(true, true, battleship);
42      yetAnotherSpecificShipCell = new SpecificShipCell(true, true, battleship);
43    }

62    public void getAttachResult() {

68    @Test
69    public void equals() {
70      Assert.assertTrue(specificShipCell.equals(specificShipCell)); // reflexivity
71      Assert.assertTrue(specificShipCell.equals(sameRefSpecificShipCell)); // trivial condition both reference the same object
72      Assert.assertTrue(specificShipCell.equals(sameStateAsSpecificShipCell)); // both objects should have the same state
73      Assert.assertTrue(sameStateAsSpecificShipCell.equals(specificShipCell)); //symmetry
74      Assert.assertEquals(specificShipCell.equals(sameStateAsSpecificShipCell) && sameStateAsSpecificShipCell.equals(yetAnotherSpecificShipCell), yetAnotherSpecificShipCell.equals(specificShipCell)); //transitivity
75      Assert.assertFalse(specificShipCell.equals(diffSpecificShipCell)); //objects have different state
76      Assert.assertFalse(specificShipCell.equals(nullSpecificShipCell)); // specificShipCell is NOT null
77    }
78  
79    @Test
80    public void testHashCode() {
81      Assert.assertEquals(specificShipCell.equals(sameRefSpecificShipCell), specificShipCell.hashCode() == sameRefSpecificShipCell.hashCode()); //same objects have the same hashCode
82      Assert.assertEquals(specificShipCell.equals(sameStateAsSpecificShipCell), specificShipCell.hashCode() == sameStateAsSpecificShipCell.hashCode()); //equal objects have the same hashCode
83    }
