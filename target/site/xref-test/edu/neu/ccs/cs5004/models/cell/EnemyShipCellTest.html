1   package edu.neu.ccs.cs5004.models.cell;
2   
3   import org.junit.Assert;
4   import org.junit.Before;
5   import org.junit.Test;
6   
7   import edu.neu.ccs.cs5004.models.attackresult.Hit;
8   import edu.neu.ccs.cs5004.models.attackresult.Miss;
9   import edu.neu.ccs.cs5004.models.attackresult.Sunk;
10  
11  public class EnemyShipCellTest {
12  
13    private EnemyShipCell enemyShipCell;
14    private EnemyShipCell enemyShipCell2;
15    private EnemyShipCell enemyShipCell3;
16    private EnemyShipCell sameRefEnemyShipCell;
17    private EnemyShipCell sameStateAsEnemyShipCell;
18    private EnemyShipCell diffEnemyShipCell;
19    private EnemyShipCell yetAnotherEnemyShipCell;
20    private EnemyShipCell nullEnemyShipCell = null;
21  
22    @Before
23    public void setUp() throws Exception {
24  
25      enemyShipCell = new EnemyShipCell(true, false);
26      enemyShipCell2 = new EnemyShipCell(true, false);
27      enemyShipCell3 = new EnemyShipCell(false, false);
28      diffEnemyShipCell = new EnemyShipCell(true, true);
29      sameRefEnemyShipCell = enemyShipCell;
30      sameStateAsEnemyShipCell = new EnemyShipCell(true, false);
31      yetAnotherEnemyShipCell = new EnemyShipCell(true, false);
32    }
33  
34    @Test
35    public void attackCell() {
36      Assert.assertEquals(enemyShipCell, enemyShipCell3.attackCell());
37    }
38  
39    @Test
40    public void markAsSunk() {
41      Assert.assertEquals(enemyShipCell2, enemyShipCell.markAsSunk());
42      Assert.assertEquals(diffEnemyShipCell, diffEnemyShipCell.markAsSunk());
43    }
44  
45    @Test
46    public void getAtteckResult() {
47      Assert.assertEquals(new Hit(), enemyShipCell.getAttackResult());
48      Assert.assertEquals(new Sunk(), diffEnemyShipCell.getAttackResult());
49      Assert.assertEquals(new Miss(), enemyShipCell3.getAttackResult());
50    }
51  
52    @Test
53    public void equals() {
54      Assert.assertTrue(enemyShipCell.equals(enemyShipCell)); // reflexivity
55      Assert.assertTrue(enemyShipCell.equals(sameRefEnemyShipCell)); // trivial condition both reference the same object
56      Assert.assertTrue(enemyShipCell.equals(sameStateAsEnemyShipCell)); // both objects should have the same state
57      Assert.assertTrue(sameStateAsEnemyShipCell.equals(enemyShipCell)); //symmetry
58      Assert.assertEquals(enemyShipCell.equals(sameStateAsEnemyShipCell) && sameStateAsEnemyShipCell.equals(yetAnotherEnemyShipCell), yetAnotherEnemyShipCell.equals(enemyShipCell)); //transitivity
59      Assert.assertFalse(enemyShipCell.equals(diffEnemyShipCell)); //objects have different state
60      Assert.assertFalse(enemyShipCell.equals(nullEnemyShipCell)); // enemyShipCell is NOT null
61    }
62  
63    @Test
64    public void testHashCode() {
65      Assert.assertEquals(enemyShipCell.equals(sameRefEnemyShipCell), enemyShipCell.hashCode() == sameRefEnemyShipCell.hashCode()); //same objects have the same hashCode
66      Assert.assertEquals(enemyShipCell.equals(sameStateAsEnemyShipCell), enemyShipCell.hashCode() == sameStateAsEnemyShipCell.hashCode()); //equal objects have the same hashCode
67    }
68  
69    @Test
70    public void testToString() {
71      Assert.assertEquals("EnemyShipCell{" +
72          "isSunk=" + enemyShipCell.getSunk() +
73          ", isHit=" + enemyShipCell.getIsHit() +
74          '}', enemyShipCell.toString());
75    }
76  }
