3   import org.junit.Assert;
4   import org.junit.Before;
5   import org.junit.Test;

7   import static org.junit.Assert.*;
8   
9   public class BattleshipTest {
10  
11    private Battleship battleship;
12    private Battleship battleship2;
13    private Battleship sameRefBattleship;
14    private Battleship sameStateAsBattleship;
15    private Battleship diffBattleship;
16    private Battleship yetAnotherBattleship;
17    private Battleship nullBattleship = null;
18  
19    @Before
20    public void setUp() throws Exception {
21  
22      battleship = new Battleship(4, 2);
23      battleship2 = new Battleship(4, 3);
24      diffBattleship = new Battleship(4, 3);
25      sameRefBattleship = battleship;
26      sameStateAsBattleship = new Battleship(4, 2);
27      yetAnotherBattleship = new Battleship(4, 2);
28    }
29  
30    @Test
31    public void hitShip() {
32      Assert.assertEquals(battleship2, battleship.hitShip());
33    }
34  
35    @Test
36    public void equals() {
37      Assert.assertTrue(battleship.equals(battleship)); // reflexivity
38      Assert.assertTrue(battleship.equals(sameRefBattleship)); // trivial condition both reference the same object
39      Assert.assertTrue(battleship.equals(sameStateAsBattleship)); // both objects should have the same state
40      Assert.assertTrue(sameStateAsBattleship.equals(battleship)); //symmetry
41      Assert.assertEquals(battleship.equals(sameStateAsBattleship) && sameStateAsBattleship.equals(yetAnotherBattleship), yetAnotherBattleship.equals(battleship)); //transitivity
42      Assert.assertFalse(battleship.equals(diffBattleship)); //objects have different state
43      Assert.assertFalse(battleship.equals(nullBattleship)); // abstractShipCell is NOT null
44    }
45  
46    @Test
47    public void testHashCode() {
48      Assert.assertEquals(battleship.equals(sameRefBattleship), battleship.hashCode() == sameRefBattleship.hashCode()); //same objects have the same hashCode
49      Assert.assertEquals(battleship.equals(sameStateAsBattleship), battleship.hashCode() == sameStateAsBattleship.hashCode()); //equal objects have the same hashCode
50    }
51    @Test
52    public void testToString() {
53      Assert.assertEquals("Battleship{" +
54          "size=" + battleship.getSize() +
55          ", hitCells=" + battleship.getHitCells() +
56          '}', battleship.toString());
57    }
58  }
