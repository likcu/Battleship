44    @Test
45    public void equals() {
46      Assert.assertTrue(abstractShipCell.equals(abstractShipCell)); // reflexivity
47      Assert.assertTrue(abstractShipCell.equals(sameRefAbstractShipCell)); // trivial condition both reference the same object
48      Assert.assertTrue(abstractShipCell.equals(sameStateAsAbstractShipCell)); // both objects should have the same state
49      Assert.assertTrue(sameStateAsAbstractShipCell.equals(abstractShipCell)); //symmetry
50      Assert.assertEquals(abstractShipCell.equals(sameStateAsAbstractShipCell) && sameStateAsAbstractShipCell.equals(yetAnotherAbstractShipCell), yetAnotherAbstractShipCell.equals(abstractShipCell)); //transitivity
51      Assert.assertFalse(abstractShipCell.equals(diffAbstractShipCell)); //objects have different state
52      Assert.assertFalse(abstractShipCell.equals(nullAbstractShipCell)); // abstractShipCell is NOT null
53      Assert.assertFalse(abstractShipCell.equals(abstractShipCell2)); // different class
54    }
55  
56    @Test
57    public void testHashCode() {
58      Assert.assertEquals(abstractShipCell.equals(sameRefAbstractShipCell), abstractShipCell.hashCode() == sameRefAbstractShipCell.hashCode()); //same objects have the same hashCode
59      Assert.assertEquals(abstractShipCell.equals(sameStateAsAbstractShipCell), abstractShipCell.hashCode() == sameStateAsAbstractShipCell.hashCode()); //equal objects have the same hashCode
60    }
