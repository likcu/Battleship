7   public class MissTest {
8   
9     private Miss miss;
10    private Miss sameRefMiss;
11    private Miss sameStateAsMiss;
12    private Miss yetAnotherMiss;
13    private Miss nullMiss = null;
14  
15    @Before
16    public void setUp() throws Exception {
17      miss = new Miss();
18      sameRefMiss = new Miss();
19      sameStateAsMiss = new Miss();
20      yetAnotherMiss = new Miss();
21    }
22  
23    @Test
24    public void equals() {
25      Assert.assertTrue(miss.equals(miss)); // reflexivity
26      Assert.assertTrue(miss.equals(sameRefMiss)); // trivial condition both reference the same object
27      Assert.assertTrue(miss.equals(sameStateAsMiss)); // both objects should have the same state
28      Assert.assertTrue(miss.equals(miss)); //symmetry
29      Assert.assertEquals(miss.equals(sameStateAsMiss) && sameStateAsMiss.equals(yetAnotherMiss), yetAnotherMiss.equals(miss)); //transitivity
30      Assert.assertFalse(miss.equals(nullMiss)); // miss is NOT null
31    }
32  
33    @Test
34    public void testHashCode() {
35      Assert.assertEquals(miss.equals(sameRefMiss), miss.hashCode() == sameRefMiss.hashCode()); //same objects have the same hashCode
36      Assert.assertEquals(miss.equals(sameStateAsMiss), miss.hashCode() == sameStateAsMiss.hashCode()); //equal objects have the same hashCode
37    }
38  
39    @Test
40    public void testToString() {
41      Assert.assertEquals("Miss{}", miss.toString());
42    }
43  }
