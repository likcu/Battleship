30    /**
31     * Determines user-defined fleet configuration under DEBUG mode.
32     *
33     * @return a ship array
34     */
35    public static Ship[] generateShips() throws IOException {
36      Message.TOTAL_SHIPNUM.printMessage();
37      String totalNumString = GameController.readLine();
38  
39      char[] totalNum = totalNumString.toCharArray();
40      int totalShips = totalNumString.equals(Message.TEN.getMessage()) ? 10 : totalNum[0] - '1' + 1;
41      while (totalShips < 1 || totalShips > 10 || (totalShips != 10 && totalNum.length != 1)) {
42        Message.TOTAL_SHIPNUM.printMessage();
43        totalNumString = GameController.readLine();
44        totalNum = totalNumString.toCharArray();
45        totalShips = totalNumString.equals(Message.TEN.getMessage()) ? 10 : totalNum[0] - '1' + 1;
46      }
47  
48      Ship[] ships = new Ship[totalShips];
49      Integer counter = 0;
50      Integer maxShipNum = 0;
51  
52      while (counter < totalShips && maxShipNum < totalShips) {
53  
54        Message.SHIPTYPE_CHOOSE.printMessage();
55        String shipType = GameController.readLine();
56  
57        while (!(shipType.equals(Message.B.getMessage()) || shipType.equals(Message.C.getMessage())
58            || shipType.equals(Message.S.getMessage()) || shipType.equals(Message.D.getMessage()))) {
59          Message.SHIPTYPE_CHOOSE.printMessage();
60          shipType = GameController.readLine();
61        }
62  
63        System.out.println(Message.SPECIFIC_SHIPNUM.getMessage() + shipType);
64        String inputString = GameController.readLine();
65        char[] inputNum = inputString.toCharArray();
66        int shipNum = inputString.equals(Message.TEN.getMessage()) ? 10 : inputNum[0] - '1' + 1;
67  
68        while (shipNum < 1 || shipNum > 10 || (shipNum != 10 && inputNum.length != 1)) {
69          Message.SHIPNUM_WARNING.printMessage();
70          System.out.println(Message.SPECIFIC_SHIPNUM + shipType);
71          inputNum = GameController.readLine().toCharArray();
72          shipNum = inputNum[0] - '1' + 1;
73        }
74  
75        maxShipNum = maxShipNum + shipNum;
76        while (maxShipNum > totalShips) {
77          System.out.println(Message.TOTAL_SHIPNUM_EXCEEDS.getMessage() + (totalShips - counter));
78          inputNum = GameController.readLine().toCharArray();
79          shipNum = inputNum[0] - '1' + 1;
80          maxShipNum = counter;
81        }
82  
83        int index = counter;
84        while (index < counter + shipNum) {
85          if (shipType.equals(Message.B.getMessage())) {
86            ships[index] = new Battleship(4, 0);
87          }
88          if (shipType.equals(Message.C.getMessage())) {
89            ships[index] = new Cruiser(3, 0);
90          }
91          if (shipType.equals(Message.S.getMessage())) {
92            ships[index] = new Submarine(2, 0);
93          }
94          if (shipType.equals(Message.D.getMessage())) {
95            ships[index] = new Destroyer(1, 0);
96          }
97          index++;
98        }
99        counter = counter + shipNum;
100       System.out.println(Message.CURRENT_SHIPNUM.getMessage() + counter);
101     }
102 
103     return ships;
104   }
